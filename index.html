怎麼用

第一步、將下面兩行貼到程式碼中


<link rel="stylesheet" href="http://js.arcgis.com/3.13/esri/css/esri.css">
<script src="http://js.arcgis.com/3.13/"></script>

第二步、載入需要用到的模組


require([
	
	// esri's map module
	"esri/map",
	
	// dojo's dom module
	"dojo/dom",
	
	// dojo plugin
	"dojo/domReady!"

], function (Map, dom){ // module name as callback function argument 


});

第三步、建立物件


require([
	"esri/map",
	"dojo/dom",
	"dojo/domReady!"
], function (Map, dom){	
	
	// Create Map Document
	var map = new Map("map", {
		center: [120.986695, 23.884165], // map center
		zoom: 8,
		basemap: "topo"	// arcgis online hosted basemap 
	});
});
